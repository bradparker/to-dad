use Term::ANSIColor;

my @words = (
  "\N{U+0068}\N{U+0061}\N{U+0070}\N{U+0070}\N{U+0079}",
  "\N{U+0062}\N{U+0069}\N{U+0072}\N{U+0074}\N{U+0068}\N{U+0064}\N{U+0061}\N{U+0079}",
  "\N{U+0064}\N{U+0061}\N{U+0064}",
  "\N{U+0079}\N{U+006F}\N{U+0075}",
  "\N{U+0072}\N{U+006F}\N{U+0063}\N{U+006B}"
);

my @colors  = ('red', 'green', 'yellow');

foreach (@words) {
  print color(@colors[int(rand(scalar @colors))]), $_ . " ";
}

print "\n";
